155void resolveKeyControls(Window* gameWindow, View* gameCamera) {
156
157 // Check all keys in map
158 for (size_t i = 0; i < engine::currentKeybindVector->size();i++) {
159 if (Keyboard::isKeyPressed(engine::currentKeybindVector->at(i).key)) {
160 // If key is pressed, run function binded to that key
161 engine::currentKeybindVector->at(i).controlCallback(gameWindow,gameCamera,engine::currentKeybindVector->at(i).args);
162 }
163 }
