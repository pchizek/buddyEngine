102 void object::setScript(XMLElement* scriptElement) {
103
104 bool scriptsToProcess = true;
105 XMLElement* nodeElement;
106 int numNodes{ 0 };
107 int numFrames{ 0 };
108
109 // Get type of script
110 if (scriptElement->Attribute("type", "move")) {
111
112 //motionInfoCache motionInfo;
113 array<int,3> motionNodesArray{ 0,0,0 };
114
115 nodeElement = scriptElement->FirstChildElement("node");
116 while (nodeElement) {
117
118 // Increase number of nodes
119 numNodes++;
120
121 // Extract node into array
122 parse(nodeElement->GetText(), motionNodesArray);
123
124 // put contents of array into vector
125 this->motionInfo.motionNodes.push_back(motionNodesArray);
126
127 // Get next node
128 nodeElement = nodeElement->NextSiblingElement("node");
129
130 }
131
132 // Enable movement
133 this->motionInfo.numNodes = numNodes;
134 this->motionInfo.active = true;
135 this->motionInfo.currentNode = 0;
136
137 //Preload first (0) node
138 this->motionInfo.dx = motionInfo.motionNodes.at(motionInfo.currentNode)[0];
139 this->motionInfo.dy = motionInfo.motionNodes.at(motionInfo.currentNode)[1];
140 this->motionInfo.dt = 1 / float(motionInfo.motionNodes.at(motionInfo.currentNode)[2]);
141
142 }
143 else if (scriptElement->Attribute("type", "animation")) {
144
145 //animInfoCache animInfo;
146 array<int, 5> animFramesArray{ 0,0,0,0,0 };
147
148 nodeElement = scriptElement->FirstChildElement("frame");
149 while (nodeElement) {
150
151 // Increase number of nodes
152 numFrames++;
153
154 // Extract node into array
155 parse(nodeElement->GetText(), animFramesArray);
156
157 // put contents of array into vector
158 this->animInfo.animFrames.push_back(animFramesArray);
159
160 // Get next node
161 nodeElement = nodeElement->NextSiblingElement("frame");
162 }
163
164 // Enable animation
165 this->animInfo.numFrames = numFrames;
166 this->animInfo.active = true;
167 this->animInfo.currentFrame = 0;
168
169 //Preload first (0) frame
170 this->objectSprite.setTextureRect(sf::IntRect(
171 animInfo.animFrames.at(animInfo.currentFrame)[0],
172 animInfo.animFrames.at(animInfo.currentFrame)[1],
173 animInfo.animFrames.at(animInfo.currentFrame)[2],
174 animInfo.animFrames.at(animInfo.currentFrame)[3]));
175
176 }

178 exception("Script: Invalid type");

188 /* Draw */
189 void object::draw(sf::RenderWindow* window) {
190
191 // Get external to the program's window
192 extern int cameraCoords[2];
193
194 // Resolve animation
195 if (animInfo.active) {
196
197 // If we've reached the next frame
198 if (animInfo.frameCounter == animInfo.animFrames.at(animInfo.currentFrame)[4] ){
199
200 // Reset frame counter
201 animInfo.frameCounter = 0;
202 animInfo.currentFrame++;
203
204 // Loop the current frame in the row
205 if (animInfo.currentFrame == animInfo.numFrames) {
206 animInfo.currentFrame = 0;
207 }
208
209 // Move actual sprite rectangle to next frame
210 objectSprite.setTextureRect(sf::IntRect(
211 animInfo.animFrames.at(animInfo.currentFrame)[0],
212 animInfo.animFrames.at(animInfo.currentFrame)[1],
213 animInfo.animFrames.at(animInfo.currentFrame)[2],
214 animInfo.animFrames.at(animInfo.currentFrame)[3]));
215
216 }
217
218 else {
219 animInfo.frameCounter++;
220 }
221
222 }
223
224 if (motionInfo.active) {
225
226 // If has reached the next node
227 if (motionInfo.nodeCounter == motionInfo.motionNodes.at(motionInfo.currentNode)[2]) {
228
229 // Reset counter, move to next node
230 motionInfo.nodeCounter = 0;
231 motionInfo.currentNode++;
232
233 // Loop the motion, Calculate steps in X and Y, and the dt
234
235 if (motionInfo.currentNode == motionInfo.numNodes){ // Reset loop
236
237 motionInfo.currentNode = 0;
238
239 motionInfo.dx = motionInfo.motionNodes.at(motionInfo.currentNode)[0];
240 motionInfo.dy = motionInfo.motionNodes.at(motionInfo.currentNode)[1];
241
242 }
243
244 else { // Next node
245
246 // Find dx, dy as path to travel to next node
247 motionInfo.dx = motionInfo.motionNodes.at(motionInfo.currentNode)[0] -
248 motionInfo.motionNodes.at(motionInfo.currentNode - 1)[0];
249 motionInfo.dy = motionInfo.motionNodes.at(motionInfo.currentNode)[1] -
250 motionInfo.motionNodes.at(motionInfo.currentNode - 1)[1];
251
252 }
253
254 motionInfo.dt = 1 / float(motionInfo.motionNodes.at(motionInfo.currentNode)[2]);
255
256 }
257
258
259 else {
260
261 // Interpolate position on path
262 motionInfo.scriptOffset[0] += motionInfo.dx * motionInfo.dt;
263 motionInfo.scriptOffset[1] += motionInfo.dy * motionInfo.dt;
264
265 motionInfo.nodeCounter++;
266
267 }
268
269 }
270
271 // Set the position of the sprite in the window
272 objectSprite.setPosition(sf::Vector2f(
273 (float)(this->worldCoords[0] - cameraCoords[0] + round(motionInfo.scriptOffset[0])),
274 (float)(this->worldCoords[1] - cameraCoords[1] + round(motionInfo.scriptOffset[1]))));
275
276 // Set it to be drawn in the next window
277 window->draw(objectSprite);
278
279 }
