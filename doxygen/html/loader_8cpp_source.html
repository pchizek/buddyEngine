32void loadAsset(XMLElement* assetElement) {
33
34 XMLElement* assetChildElement;
35
36 assetChildElement = assetElement->FirstChildElement("file");
37 const char* fileName = assetChildElement->GetText();
38
39 // Get filename
40 const char* aliasName = assetElement->Attribute("alias");
41 string assetKey;
42 if (aliasName) {
43 assetKey = string(aliasName);
44 }
45 else {
46 assetKey = string(fileName);
47 assetKey = assetKey.substr(0,assetKey.find("."));
48 }
49 string texturePath = string(ASSET_PATH).append(fileName);
50
51 // Get asset scale
52 float assetScaleArray[2];
53 assetChildElement = assetChildElement->NextSiblingElement("scale");
54 if (assetChildElement){
55 parse(assetChildElement->GetText(), assetScaleArray, 2);
56 }
57 else {
58 assetScaleArray[0] = 1.0f;
59 assetScaleArray[1] = 1.0f;
60 }
61 sf::Vector2f scaleFactor;
62 scaleFactor.x = assetScaleArray[0];
63 scaleFactor.y = assetScaleArray[1];
64
65
66 // Call function to import the texture
67 engine::importTexture(texturePath, assetKey, scaleFactor);
68
69}

71void loadObject(XMLElement* objectElement) {
72
73 // Access the unordered map of textures
74 XMLElement* objectChildElement;
75
76 /* World location under <grid> or <coords> */
77 int worldLoc[2];
78
79 if (objectElement->FirstChildElement("coords")) {
80
81 objectChildElement = objectElement->FirstChildElement("coords");
82
83 // Extract from document
84 parse(objectChildElement->GetText(),worldLoc,2);
85
86 }
87 else if (objectElement->FirstChildElement("grid")){
88
89 objectChildElement = objectElement->FirstChildElement("grid");
90
91 // Extract from document
92 parse(objectChildElement->GetText(),worldLoc,2);
93
94 worldLoc[0] *= BLOCK_SIZE;
95 worldLoc[1] *= BLOCK_SIZE;
96
97 }
98 else {
99 exception("Object: Invalid location");
100 }
101
102 /* Layer */
103 objectChildElement = objectElement->FirstChildElement("layer");
104 int objLayer{};
105
106 if (objectChildElement) {
107
108 parse(objectChildElement->GetText(), &objLayer);
109 if ((objLayer >= MAX_LAYERS) || (objLayer < 0)) exception("Object: Invalid layer value");
110
111 }
112 else {
113 exception("Object: Missing layer");
114 }
115
116 /* Texture */
117 objectChildElement = objectElement->FirstChildElement("texture");
118 textureInfo* objectTextureInfo = NULL;
119
120 if (objectChildElement) {
121
122 // Get the asset from the map
123 unordered_map<string, textureInfo>::iterator asset = engine::textureInfoMap.find(objectChildElement->GetText());
124
125 // Check if asset is valid
126 if (asset == engine::textureInfoMap.end()) {
127 exception("Object: Texture not found");
128 }
129 else {
130 objectTextureInfo = &asset->second;
131 }
132
133 }
134 else {
135 exception("Object: Invalid texture");
136 }
137
138 /* Block selection or rectangle */
139 int spriteRect[4] = {0,0,0,0};
140
141 if (objectElement->FirstChildElement("rect")) {
142
143 objectChildElement = objectElement->FirstChildElement("rect");
144
145 // Extract from document
146 parse(objectChildElement->GetText(), spriteRect, 4);
147
148 }
149
150 else if (objectElement->FirstChildElement("blockSelect")) {
151 int blockSelect;
152
153 objectChildElement = objectElement->FirstChildElement("blockSelect");
154
155 // Extract from document
156 parse(objectChildElement->GetText(), &blockSelect);
157
158 // Set texture rectangle
159 spriteRect[0] = (32 * blockSelect);
160 spriteRect[1] = 0;
161 spriteRect[2] = 32;
162 spriteRect[3] = 32;
163
164 }
165
166 else {
167
168 // Set texture rectangle
169 spriteRect[0] = 0;
170 spriteRect[1] = 0;
171
172 // Get size of rectangle as full size of texture
173 sf::Vector2u fullSize = objectTextureInfo->assetTexture.getSize(); // Dereferences null pointer but could null pointer ever get here?
174 spriteRect[2] = fullSize.x;
175 spriteRect[3] = fullSize.y;
176 }
177
178
179
180 sf::IntRect sRect(spriteRect[0], spriteRect[1], spriteRect[2], spriteRect[3]);
181
182
183 /* Construct object */
184 engine::object* newObject = new engine::object(worldLoc, objLayer, objectTextureInfo, &sRect);
185
186 /* Set object scale factor */
187 float localScaleFactorArr[2];
188 sf::Vector2f localScaleFactor;
189 if (objectElement->FirstChildElement("scale")) {
190 objectChildElement = objectElement->FirstChildElement("scale");
191 parse(objectChildElement->GetText(), localScaleFactorArr, 2);
192 localScaleFactor.x = localScaleFactorArr[0];
193 localScaleFactor.y = localScaleFactorArr[1];
194 }
195 else {
196 localScaleFactor.x = 1.0f;
197 localScaleFactor.y = 1.0f;
198 }
199
200 newObject->setLocalScaleFactor(&localScaleFactor);
201 newObject->setObjectScaleFactor();
202
203 /*
204 * Add scripts
205 * (these may or may not apply to a given object so they
206 * have to be added to the object already constructed)
207 */
208 if (objectElement->FirstChildElement("script")) {
209 objectChildElement = objectElement->FirstChildElement("script");
210 newObject->setScript(objectChildElement);
211 }
212
213 /* Emplace object */
214 newObject->prevObject = engine::lastObject;
215 if (engine::firstObject == NULL) {
216 engine::firstObject = newObject;
217 }
218 else {
219 newObject->prevObject->nextObject = newObject;
220 }
221 //newObject->prevObject = engine::lastObject;
222
223 engine::lastObject = newObject;
224
225}

230void loadControlScheme() {
231
232 XMLDocument controlSchemesDoc;
233 XMLError loadError = controlSchemesDoc.LoadFile("resources/ui/controlSchemes.xml");
234
235 // Throw error
236 if (loadError) {
237 exception("Controls: Cannot open document");
238 }
239
240 XMLElement* schemeElement = controlSchemesDoc.FirstChildElement("scheme");
241 XMLElement* schemeChildElement;
242
243 while (schemeElement) {
244
245 const char* schemeName = schemeElement->Attribute("name");
246 string schemeKey;
247 if (schemeName) {
248 schemeKey = string(schemeName);
249 }
250 else { exception("Error reading control scheme"); }
251
252 // Get keybinds
253 schemeChildElement = schemeElement->FirstChildElement("keybind");
254
255 keybindVector thisKeybindVector;
256
257 while (schemeChildElement) {
258
259 keyControlFunction thisKeyFunction;
260
261 // Get key on keyboard
262 thisKeyFunction.key = convertToKey.at(string(schemeChildElement->FirstChildElement("key")->GetText()));
263
264 // Get function callback by name
265 string functionName = string(schemeChildElement->FirstChildElement("function")->GetText());
266 thisKeyFunction.controlCallback = engine::controlFunctionMap.at(functionName);
267
268 // Get args
269 parse(string(schemeChildElement->FirstChildElement("args")->GetText()), thisKeyFunction.args);
270
271 // Put this key bind in the vector
272 thisKeybindVector.push_back(thisKeyFunction);
273
274 // Next keybind in scheme
275 schemeChildElement = schemeChildElement->NextSiblingElement("keybind");
276 }
277
278 // Put this keybind vector into map of all keybind schemes
279 controlSchemeMap.emplace(schemeKey, thisKeybindVector);
280
281 // Next scheme
282 schemeElement = schemeElement->NextSiblingElement("scheme");
283 }
284}
